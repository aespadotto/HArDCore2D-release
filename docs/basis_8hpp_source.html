640  compute(

661  template<typename BasisType>
662  static boost::multi_array<typename detail::basis_evaluation_traits<Family<BasisType>, BasisFunction>::ReturnValue, 2>
663  compute(
664  const Family<BasisType> & basis,
665  const QuadratureRule & quad
666  )
667  {
668  typedef detail::basis_evaluation_traits<Family<BasisType>, BasisFunction> traits;
669 
670  boost::multi_array<typename traits::ReturnValue, 2>
671  basis_quad( boost::extents[basis.dimension()][quad.size()] );
672 
673  // Compute values at quadrature note on ancestor basis, and then do a transformation
674  const auto & ancestor_basis = basis.ancestor();
675  boost::multi_array<typename traits::ReturnValue, 2>
676  ancestor_basis_quad = evaluate_quad<BasisFunction>::compute(ancestor_basis, quad);
677 
678  Eigen::MatrixXd M = basis.matrix();
679  for (size_t i = 0; i < size_t(M.rows()); i++) {
680  for (size_t iqn = 0; iqn < quad.size(); iqn++) {
681  basis_quad[i][iqn] = M(i,0) * ancestor_basis_quad[0][iqn];
682  }
683  for (size_t j = 1; j < size_t(M.cols()); j++) {
684  for (size_t iqn = 0; iqn < quad.size(); iqn++) {
685  basis_quad[i][iqn] += M(i,j) * ancestor_basis_quad[j][iqn];
686  }
687  } // for j
688  } // for i
689 
690  return basis_quad;
691  }

704  template<typename T>
705  Eigen::MatrixXd gram_schmidt(
706  boost::multi_array<T, 2> & basis_eval,
707  const std::function<double(size_t, size_t)> & inner_product
708  )
709  {
710  auto induced_norm = [inner_product](size_t i)->double
711  {
712  return std::sqrt( inner_product(i, i) );
713  };
714 
715  // Number of basis functions
716  size_t Nb = basis_eval.shape()[0];
717  // Number of quadrature nodes
718  size_t Nqn = basis_eval.shape()[1];
719 
720  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(Nb, Nb);
721 
722  // Normalise the first element
723  double norm = induced_norm(0);
724  for (size_t iqn = 0; iqn < Nqn; iqn++) {
725  basis_eval[0][iqn] /= norm;
726  } // for iqn
727  B(0,0) = 1./norm;
728 
729  for (size_t ib = 1; ib < Nb; ib++) {
730  // 'coeffs' represents the coefficients of the ib-th orthogonal function on the ON basis functions 0 to ib-1
731  Eigen::RowVectorXd coeffs = Eigen::RowVectorXd::Zero(ib);
732  for (size_t pb = 0; pb < ib; pb++) {
733  coeffs(pb) = - inner_product(ib, pb);
734  } // for pb
735 
736  // store the values of the orthogonalised ib-th basis function
737  for (size_t pb = 0; pb < ib; pb++) {
738  for (size_t iqn = 0; iqn < Nqn; iqn++) {
739  basis_eval[ib][iqn] += coeffs(pb) * basis_eval[pb][iqn];
740  } // for iqn
741  } // for pb
742 
743  // normalise ib-th basis function
744  double norm = induced_norm(ib);
745  for (size_t iqn = 0; iqn < Nqn; iqn++) {
746  basis_eval[ib][iqn] /= norm;
747  } // for iqn
748  coeffs /= norm;
749  // Compute ib-th row of B.
750  // B.topLeftCorner(ib, ib) contains the rows that represent the ON basis functions 0 to ib-1 on
751  // the original basis. Multiplying on the left by coeffs gives the coefficients of the ON basis
752  // functions on the original basis functions from 0 to ib-1.
753  B.block(ib, 0, 1, ib) = coeffs * B.topLeftCorner(ib, ib);
754  B(ib, ib) = 1./norm;
755  }
756  return B;
757  }

776  template<typename BasisType>
777  Family<BasisType> l2_orthonormalize(
778  const BasisType & basis,
779  const QuadratureRule & qr,
780  boost::multi_array<typename BasisType::FunctionValue, 2> & basis_quad
781  )
782  {
783  // Check that the basis evaluation and quadrature rule are coherent
784  assert ( basis.dimension() == basis_quad.shape()[0] && qr.size() == basis_quad.shape()[1] );
785 
786  // The inner product between the i-th and j-th basis vectors
787  std::function<double(size_t, size_t)> inner_product
788  = [&basis_quad, &qr](size_t i, size_t j)->double
789  {
790  double r = 0.;
791  for(size_t iqn = 0; iqn < qr.size(); iqn++) {
792  r += qr[iqn].w * scalar_product(basis_quad[i][iqn], basis_quad[j][iqn]);
793  } // for iqn
794  return r;
795  };
796 
797  Eigen::MatrixXd B = gram_schmidt(basis_quad, inner_product);
798 
799  return Family<BasisType>(basis, B);
800  }

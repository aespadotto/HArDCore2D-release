HArD::Core2D
Hybrid Arbitrary Degree::Core 2D - Library to implement 2D schemes with edge and cell polynomials as unknowns
  • src

hybridcore.hpp
1 // Core data structures and methods required to implement hybrid schemes in 2D (polynomial unknowns
2 // in the cells and on the edges, such as Hybrid High-order (HHO) schemes).
3 //
4 // Provides:
5 // - Hybrid polynomial basis functions (on the cells and edges of the mesh)
6 // - Generic routines to create quadrature nodes over cells and edges of the mesh
7 // - Interpolation of general functions onto the HHO space
8 // - Methods for integrating, evaluating, and computing norms of HHO solutions
9 //
10 // Author: Jerome Droniou (jerome.droniou@monash.edu)
11 //
12 
13 /*
14 *
15 * This library was developed around HHO methods, although some parts of it have a more
16 * general purpose. If you use this code or part of it in a scientific publication,
17 * please mention the following book as a reference for the underlying principles
18 * of HHO schemes:
19 *

313 template <typename Function>
314 void HybridCore::quadrature_over_cell(const size_t iT, const Function& f) const {
315  assert(iT < _mesh_ptr->n_cells());
316  Cell* cell = _mesh_ptr->cell(iT);
317  QuadratureRule quadT = generate_quadrature_rule(*cell, 2 * _Ldeg + 5);
318  for (size_t iqn = 0; iqn < quadT.size(); iqn++) {
319  f(iqn, quadT[iqn].x, quadT[iqn].y, quadT[iqn].w);
320  }
321 }
322 
323 template <typename Function>
324 void HybridCore::quadrature_over_edge(const size_t iF, const Function& f) const {
325  assert(iF < _mesh_ptr->n_edges());
326  Edge* edge = _mesh_ptr->edge(iF);
327  QuadratureRule quadF = generate_quadrature_rule(*edge, 2 * _K + 5);
328  for (size_t iqn = 0; iqn < quadF.size(); iqn++) {
329  f(iqn, quadF[iqn].x, quadF[iqn].y, quadF[iqn].w);
330  }
331 }

333 template <typename Function>
334 double HybridCore::integrate_over_cell(const size_t iT, const Function& f) const {
335  assert(iT < _mesh_ptr->n_cells());
336  double ans = 0.0;
337  quadrature_over_cell(iT, [&ans, &f](size_t iQN, double x, double y,
338  double w) { ans += w * f(x, y); });
339  return ans;
340 }

367 template<typename Function>
368 Eigen::VectorXd HybridCore::interpolate(const Function& f, size_t doe) const {
369  Eigen::VectorXd XTF = Eigen::VectorXd::Zero(_ntotal_dofs);
370 
371  // Projections on faces
372  for (size_t iF = 0; iF < _mesh_ptr->n_edges(); iF++) {
373  Edge* edge = _mesh_ptr->edge(iF);
374 
375  // Mass matrix on edge
376  QuadratureRule quadF = generate_quadrature_rule(*edge, doe);
377  std::vector<Eigen::ArrayXd> phiF_quadF = basis_quad("edge", iF, quadF, _K);
378  Eigen::MatrixXd MF = gram_matrix(phiF_quadF, phiF_quadF, _nlocal_edge_dofs, _nlocal_edge_dofs, quadF, true);
379 
380  // Vector of integral of f against edge basis functions
381  Eigen::VectorXd bF = Eigen::VectorXd::Zero(_nlocal_edge_dofs);
382  for (size_t i = 0; i < _nlocal_edge_dofs; i++) {
383  for (size_t iqn = 0; iqn < quadF.size(); iqn++){
384  bF(i) += quadF[iqn].w * phiF_quadF[i](iqn) * f(quadF[iqn].x, quadF[iqn].y);
385  }
386  }
387 
388  // Vector of coefficients (on cell basis functions) of the L2(F) projection of f
389  Eigen::VectorXd UF = MF.ldlt().solve(bF);
390 
391  // Fill in he complete vector of DOFs
392  XTF.segment(_ntotal_cell_dofs + iF * _nlocal_edge_dofs, _nlocal_edge_dofs) = UF;
393  }
394 
395  // Projections on cells
396  for (size_t iT = 0; iT < _mesh_ptr->n_cells(); iT++) {
397  Cell* cell = _mesh_ptr->cell(iT);
398  size_t nedges = cell->n_edges();
399 
400  // Mass matrix in cell
401  QuadratureRule quadT = generate_quadrature_rule(*cell, doe);
402  std::vector<Eigen::ArrayXd> phiT_quadT = basis_quad("cell", iT, quadT, _Ldeg);
403  Eigen::MatrixXd MT = gram_matrix(phiT_quadT, phiT_quadT, _nlocal_cell_dofs, _nlocal_cell_dofs, quadT, true);
404 
405  // Vector of integrals of f against basis functions
406  Eigen::VectorXd bT = Eigen::VectorXd::Zero(_nlocal_cell_dofs);
407  for (size_t i = 0; i < _nlocal_cell_dofs; i++) {
408  for (size_t iqn = 0; iqn < quadT.size(); iqn++){
409  bT(i) += quadT[iqn].w * phiT_quadT[i](iqn) * f(quadT[iqn].x, quadT[iqn].y);
410  }
411  }
412 
413  // Vector of coefficients (on cell basis functions) of the L2(T) projection of f
414  Eigen::VectorXd UT = MT.ldlt().solve(bT);
415 
416  // Fill in the complete vector of DOFs
417  size_t offset_T = iT * _nlocal_cell_dofs;
418  XTF.segment(offset_T, _nlocal_cell_dofs) = UT;
419 
420  // Special case of L=-1, we replace the previously computed cell value with the proper average of edge values
421  if ( _L==-1 ){
422  // Weights corresponding to the values in cells/on edges
423  Eigen::VectorXd barycoefT = compute_weights(iT);
424  // We transform these weights so that they apply to the coefficients on the basis functions
425  Eigen::Vector2d xT = cell->center_mass();
426  double phiT_cst = cell_basis(iT,0)(xT.x(), xT.y());
427  for (size_t ilF = 0; ilF < nedges; ilF++){
428  Eigen::Vector2d xF = cell->edge(ilF)->center_mass();
429  size_t iF = cell->edge(ilF)->global_index();
430  double phiF_cst = edge_basis(iF,0)(xF.x(), xF.y());
431  barycoefT(ilF) *= phiF_cst / phiT_cst;
432  }
433 
434  XTF(iT) = 0;
435  for (size_t ilF = 0; ilF < nedges; ilF++) {
436  size_t iF = cell->edge(ilF)->global_index();
437  XTF(iT) += barycoefT(ilF) * XTF(_ntotal_cell_dofs + iF);
438  }
439  }
440 
441  }
442 
443  return XTF;
444 }
